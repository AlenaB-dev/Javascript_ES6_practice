ğŸ“¦ Common Sorting Algorithms

1. Bubble Sort

  • Idea: Repeatedly swap adjacent elements if they are in the wrong order.
  • Time Complexity: O(nÂ²)
  • Simple but inefficient for large arrays
function bubbleSort(arr) {
  let n = arr.length;
  for (let i = 0; i < n; i++) {
    for (let j = 0; j < n - 1 - i; j++) {
      if (arr[j] > arr[j + 1]) {
        // Swap
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
      }
    }
  }
  return arr;
}

2. Selection Sort

  • Idea: Find the smallest item and put it in the correct position.
  • Time Complexity: O(nÂ²)
function selectionSort(arr) {
  let n = arr.length;
  for (let i = 0; i < n; i++) {
    let min = i;
    for (let j = i + 1; j < n; j++) {
      if (arr[j] < arr[min]) {
        min = j;
      }
    }
    [arr[i], arr[min]] = [arr[min], arr[i]];
  }
  return arr;
}

3. Insertion Sort

  • Idea: Build a sorted array one item at a time.
  • Time Complexity: O(nÂ²)
  • Efficient for small or nearly-sorted arrays
function insertionSort(arr) {
  for (let i = 1; i < arr.length; i++) {
    let value = arr[i];
    let j = i - 1;
    while (j >= 0 && arr[j] > value) {
      arr[j + 1] = arr[j];
      j--;
    }
    arr[j + 1] = value;
  }
  return arr;
}

4. Merge Sort

  • Idea: Divide and conquer. Split the array, sort each half, and merge.
  • Time Complexity: O(n log n)
  • Stable and efficient
function mergeSort(arr) {
  if (arr.length <= 1) return arr;

  const mid = Math.floor(arr.length / 2);
  const left = mergeSort(arr.slice(0, mid));
  const right = mergeSort(arr.slice(mid));

  return merge(left, right);
}

function merge(left, right) {
  let result = [], i = 0, j = 0;

  while (i < left.length && j < right.length) {
    result.push(left[i] < right[j] ? left[i++] : right[j++]);
  }

  return result.concat(left.slice(i)).concat(right.slice(j));
}

5. Quick Sort

  • Idea: Pick a pivot, partition array, recursively sort the parts.
  • Time Complexity: O(n log n) on average
  • Can be very fast, but not stable
function quickSort(arr) {
  if (arr.length <= 1) return arr;

  const pivot = arr[arr.length - 1];
  const left = [], right = [];

  for (let i = 0; i < arr.length - 1; i++) {
    arr[i] < pivot ? left.push(arr[i]) : right.push(arr[i]);
  }

  return [...quickSort(left), pivot, ...quickSort(right)];
}
